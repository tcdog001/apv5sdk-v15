#ifndef HAVECLASS_SOURCE
#define HAVECLASS_SOURCE

#include "../scan/scan.h"

bool haveclass (SCAN * content, const char *charset) 

{
	if (isclass (content, charset)) 
	{
		nexttoken (content);
		return (true);
	}
	return (false);
}

#endif
