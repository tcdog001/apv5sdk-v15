#ifndef DECODESETKEY_SOURCE
#define DECODESETKEY_SOURCE

#include <stdint.h>
#include <errno.h>

#include "../ihpapi/ihp.h"
 
int ihp_DecodeSetKey (const uint8_t buffer [], size_t length, ihpapi_result_t * result) 

{
	struct vs_set_key_cnf 
	{
		struct header_vs header;
		int8_t MSTATUS;
	}
	*confirm = (struct vs_set_key_cnf *)(buffer);
	if (length < sizeof (struct vs_set_key_cnf)) 
	{
		errno = ERANGE;
		return (-1);
	}
	if (result == (ihpapi_result_t *)(0)) 
	{
		errno = EFAULT;
		return (-1);
	}
	result->opStatus.status = confirm->MSTATUS;
	return (0);
}

#endif
