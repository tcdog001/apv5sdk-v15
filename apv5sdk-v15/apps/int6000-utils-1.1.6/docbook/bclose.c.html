#ifndef BCLOSE_SOURCE
#define BCLOSE_SOURCE

#include <errno.h>
#include <stdint.h>
#include <stdlib.h>

#include "../tools/buffer.h"

void bclose (buffer_t * b) 

{
	if (b != 0) 
	{
		free (b);
	}
	return;
}
